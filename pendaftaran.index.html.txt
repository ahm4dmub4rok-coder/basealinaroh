'bg-primary-600 text-white' : 'border border-primary-200 text-primary-600 hover:bg-primary-50'} transition-all">${i}</button>`; else if (i === currentSantriPage - 2 || i === currentSantriPage + 2) h += '<span class="px-2 text-primary-400">...</span>'; }
            h += `<button onclick="changePage(${currentSantriPage + 1})" ${currentSantriPage === tp ? 'disabled' : ''} class="px-3 py-1.5 rounded-lg border border-primary-200 text-primary-600 hover:bg-primary-50 disabled:opacity-50 disabled:cursor-not-allowed transition-all"><i class="fas fa-chevron-right"></i></button>`;
            document.getElementById('pagination').innerHTML = h;

        function saveSantriData() {
            const form = document.getElementById('santriForm');
            const formData = new FormData(form);
            const santriList = getSantri();
            
            const requiredFields = ['nis', 'namaLengkap', 'tanggalDaftar', 'tanggalMasuk', 'tanggalLahir', 'statusDomisili', 'kelas', 'status'];
            for (let field of requiredFields) {
                if (!formData.get(field)) {
                    showToast(`Field ${field} wajib diisi!`, 'error');
                    return;
                }
            }

            const data = {
                nis: formData.get('nis') || '', noDaftar: formData.get('noDaftar') || '', tanggalDaftar: formData.get('tanggalDaftar') || '', tanggalMasuk: formData.get('tanggalMasuk') || '',
                namaLengkap: formData.get('namaLengkap') || '', namaAkte: formData.get('namaAkte') || '', tempatLahir: formData.get('tempatLahir') || '', tanggalLahir: formData.get('tanggalLahir') || '',
                nik: formData.get('nik') || '', nisn: formData.get('nisn') || '', kk: formData.get('kk') || '',
                desa: formData.get('desa') || '', kecamatan: formData.get('kecamatan') || '', kabupaten: formData.get('kabupaten') || '', provinsi: formData.get('provinsi') || '',
                namaAyah: formData.get('namaAyah') || '', hpAyah: formData.get('hpAyah') || '',
                namaIbu: formData.get('namaIbu') || '', hpIbu: formData.get('hpIbu') || '',
                namaWali: formData.get('namaWali') || '', hpWali: formData.get('hpWali') || '',
                statusDomisili: formData.get('statusDomisili') || '', asrama: formData.get('asrama') || '',
                kelas: formData.get('kelas') || '', absen: formData.get('absen') || '', ruangKelas: formData.get('ruangKelas') || '',
                status: formData.get('status') || '', keterangan: formData.get('keterangan') || '', foto: ''
            };

            const exists = santriList.find(s => s.nis === data.nis && s.id !== editingSantriId);
            if (exists) { showToast('NIS sudah terdaftar!', 'error'); return; }

            if (editingSantriId) {
                const idx = santriList.findIndex(s => s.id === editingSantriId);
                if (idx !== -1) {
                    santriList[idx] = { ...santriList[idx], ...data, id: editingSantriId, created: santriList[idx].created, updated: new Date().toISOString() };
                    addActivity(`Mengupdate data santri: ${data.namaLengkap}`, 'update');
                    addAuditLog('UPDATE_SANTRI', `Data ${data.namaLengkap} diperbarui`);
                    showToast('Data berhasil diperbarui!');
                }
            } else {
                santriList.push({ id: Date.now(), ...data, created: new Date().toISOString(), updated: new Date().toISOString() });
                addActivity(`Menambahkan santri baru: ${data.namaLengkap}`, 'create');
                addAuditLog('CREATE_SANTRI', `Santri baru ${data.namaLengkap} ditambahkan`);
                showToast('Santri berhasil ditambahkan!');
            }

            localStorage.setItem('sis_santri', JSON.stringify(santriList));
            resetForm();
            showPage('dataSantri');
        }

        function editSantri(id) {
            const s = getSantri().find(item => item.id === id);
            if (s) {
                editingSantriId = id;
                const form = document.getElementById('santriForm');
                form.reset();
                form.querySelectorAll('input, select, textarea').forEach(field => {
                    if (s[field.name] !== undefined) field.value = s[field.name] || '';
                });
                showPage('tambahSantri');
                document.getElementById('pageTitle').textContent = 'Edit Santri';
                document.getElementById('pageSubtitle').textContent = 'Edit data santri';
            }
        }

        function deleteSantri(id) {
            showConfirmModal('Hapus Data Santri', 'Apakah Anda yakin ingin menghapus data santri ini?', () => {
                // Logic hapus
                let list = getSantri();
                const s = list.find(i => i.id === id);
                list = list.filter(i => i.id !== id);
                localStorage.setItem('sis_santri', JSON.stringify(list));
                
                // Logging
                addActivity(`Menghapus data: ${s ? s.namaLengkap : id}`, 'delete');
                addAuditLog('DELETE_SANTRI', `Santri ${s ? s.namaLengkap : id} dihapus`);
                
                // UI Feedback
                showToast('Data berhasil dihapus!');
                loadSantriTable(); // Refresh table
            });
        }

        function restoreData(input) {
            if (input.files && input.files[0]) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    try {
                        const d = JSON.parse(e.target.result);
                        if (d.santri) localStorage.setItem('sis_santri', JSON.stringify(d.santri));
                        if (d.users) localStorage.setItem('sis_users', JSON.stringify(d.users));
                        showToast('Data berhasil di-restore!');
                        setTimeout(() => location.reload(), 1000);
                    } catch { showToast('File tidak valid!', 'error'); }
                };
                reader.readAsText(input.files[0]);
            }
        }
